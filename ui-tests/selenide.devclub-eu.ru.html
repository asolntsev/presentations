Решения: Тамагочи

Искать элементы по тексту 

import static com.codeborne.selenide.Selectors.*;

$(byText("Привет, чертяка!"))
   .shouldBe(visible);

$(withText("ертяк"))
   .shouldHave(text("Привет, чертяка"));
